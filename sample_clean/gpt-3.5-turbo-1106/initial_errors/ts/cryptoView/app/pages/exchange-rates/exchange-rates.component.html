class CurrencyRateCardComponent { @Input() selected: any; search: FormControl = new FormControl(); filteredOptions: Observable; constructor() { this.filteredOptions = this.search.valueChanges.pipe( startWith(''), map(value => this._filter(value)) ); } private _filter(value: string): string[] { const filterValue = value.toLowerCase(); return this.options.filter(option => option.toLowerCase().includes(filterValue)); } }
